async function login() { const email = document.getElementById('email').value.trim(); const password = document.getElementById('password').value.trim(); const errorDiv = document.getElementById('errorMsg'); const loadingDiv = document.getElementById('loadingMsg'); errorDiv.style.display = 'none'; if (!email || !password) { errorDiv.innerHTML = '❌ الرجاء إدخال البريد الإلكتروني وكلمة المرور'; errorDiv.style.display = 'block'; return; } loadingDiv.style.display = 'block'; try { const response = await fetch('/api/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ email, password }) }); const data = await response.json(); loadingDiv.style.display = 'none'; if (data.success && data.user) { localStorage.setItem('currentUser', JSON.stringify(data.user)); // التوجيه حسب الدور (role) if (data.user.role === 'admin') { window.location.href = '/admin.html'; } else { window.location.href = '/dashboard.html'; } } else { errorDiv.innerHTML = '❌ ' + (data.error || 'بيانات الدخول غير صحيحة'); errorDiv.style.display = 'block'; } } catch (err) { loadingDiv.style.display = 'none'; errorDiv.innerHTML = '❌ خطأ في الاتصال: ' + err.message; errorDiv.style.display = 'block'; } }
